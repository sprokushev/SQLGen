Select --distinct
t.relname,
--    c.column_order as field_order,
--    c.column_name as field_name,
--    col_description(t.oid, c.column_order) as field_desc,
  	c.data_type as field_type,
    case 
        when c.data_type = 'BIT' or c.data_type = 'CHAR' or c.data_type = 'VARCHAR' then cast(c.maximum_length as varchar)
        when c.data_type = 'NUMERIC' then cast(c.numeric_precision as varchar)
        else ''
    end as field_size,
    case 
        when c.data_type = 'NUMERIC' then cast(c.numeric_scale as varchar)
        else ''
    end as field_dec,
    c.*
    /*
    
    case when c.is_nullable = 'YES' then 'false' else 'true' end as IsNotNull,
    case when c.is_identity = 'YES' then 'true' else 'false' end as IsIdentity,
    c.column_default as field_default,
    */
FROM pg_catalog.pg_class t
INNER JOIN pg_namespace n ON n.oid = t.relnamespace
INNER JOIN LATERAL (
	SELECT 
    	attr.attnum as column_order,
        attr.attname as column_name,
        upper(case 
    		when typ.typname = 'character' or typ.typname = '""char""' or typ.typname = 'bpchar' then 'char'
		    when typ.typname = 'character varying' then 'varchar'
		    when typ.typname = 'int8' then 'bigint'
		    when typ.typname = 'serial8' then 'bigserial'
		    when typ.typname = 'bit varying' then 'varbit'
		    when typ.typname = 'float8' then 'double precision'
		    when typ.typname = 'int4' then 'int'
		    when typ.typname = 'float4' then 'real'
		    when typ.typname = 'int2' then 'smallint'
		    when typ.typname = 'serial4' then 'serial'
            when typ.typname = 'timestamptz' then 'timestamp with time zone'
		    else typ.typname
		  end) as data_type,
          CASE atttypid
         	WHEN 21 /*int2*/ THEN 16
         	WHEN 23 /*int4*/ THEN 32
         	WHEN 20 /*int8*/ THEN 64
         	WHEN 1700 /*numeric*/ THEN
              CASE WHEN atttypmod = -1 THEN null
                   ELSE ((atttypmod - 4) >> 16) & 65535     -- calculate the precision
              END
         	WHEN 700 /*float4*/ THEN 24 /*FLT_MANT_DIG*/
         	WHEN 701 /*float8*/ THEN 53 /*DBL_MANT_DIG*/
         	ELSE null
	 	  END AS numeric_precision,
		  CASE 
		    WHEN atttypid IN (21, 23, 20) THEN 0
		    WHEN atttypid IN (1700) THEN            
	        	CASE 
    	        	WHEN atttypmod = -1 THEN null       
	        	    ELSE (atttypmod - 4) & 65535            -- calculate the scale  
		        END
    		ELSE null
		  END AS numeric_scale,
          CASE WHEN atttypmod = -1 THEN null ELSE (atttypmod - 4) END as maximum_length,
          attr.*
    FROM pg_catalog.pg_attribute attr
    INNER JOIN pg_catalog.pg_type typ ON typ.oid = attr.atttypid
    LEFT JOIN pg_catalog.pg_attrdef def ON def.adrelid = t.oid AND def.adnum = attr.attnum
    WHERE attr.attrelid = t.oid
	AND attr.attisdropped = false -- не удалено
    AND attr.attnum > 0
) c ON true
WHERE t.relkind in ('r', 'f') -- таблицы
AND n.nspname ilike 'dbo'
--AND t.relname ilike 'evndirection'
AND c.column_name not like '%_rowversion'
AND c.data_type not in ('BIGINT', 'CHAR', 'VARCHAR', 'INT', 'XML', 'TIMESTAMP', 'TEXT', 'DATE', 'UUID', 'SMALLINT', 'TIME', 'TIMESTAMP WITH TIME ZONE',
'DOUBLE PRECISION', 'NUMERIC', 'BOOL', 'JSONB', 'JSON', 'REAL')
--order by c.column_order
limit 100
